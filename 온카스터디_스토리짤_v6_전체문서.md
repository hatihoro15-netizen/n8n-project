## 장면 구성 규칙
- 총 3개 장면
- 모든 장면이 두 사람 함께 등장 (3개 전부)
- AB 함께 장면에서는 speaker를 말하는 사람으로 설정
- 각 장면의 대사는 2~3문장 (8초 영상에 맞는 분량, 한 장면당 30~50자)
- 한 영상의 모든 장면은 동일한 하나의 장소에서 진행
- 장면마다 포즈, 행동, 표정, 카메라 앵글은 다르게 변화를 줄 것

## 배경/장소 규칙 (★ 매우 중요)

**한 영상의 모든 장면은 동일한 하나의 장소에서 진행된다.**
- 3개 장면 전부 같은 장소, 같은 배경
- 장면마다 달라지는 것: 캐릭터의 포즈, 행동, 표정, 소품 사용, 카메라 앵글
- 그 장소에서 자연스러운 소품, 행동, 상호작용 포함

**매 영상(에피소드)마다 완전히 새로운 장소를 직접 창작할 것.**
- 이전 영상에서 사용한 장소와 절대 겹치지 않는 새로운 상황을 만들어라
- 참고 리스트 없음. 현실에서 친구 둘이 있을 법한 일상 상황을 직접 상상해서 창작해라
- 그 장소에서 자연스러운 소품, 행동, 상호작용을 포함해라
- 뻔한 장소 반복 금지. 매번 새롭고 예측 불가능한 장소가 핵심이다

### 8. NCA 데이터 준비 (FFmpeg concat + SRT)
```javascript
// NCA 스토리형 숏츠 - Veo3 Fast (audio on) - 가로(16:9) 버전
const hookTitle = $('콘텐츠 파싱').first().json.hook_title;
const segmentCount = $('콘텐츠 파싱').first().json.segmentCount;

// === 세그먼트 데이터 수집 ===
let segments = [];
let currentStart = 0;
const CLIP_DURATION = 8; // Veo3 8초 클립

for (let i = 0; i < segmentCount; i++) {
  const videoResult = $('영상 완료 체크').all()[i]?.json;
  let videoUrl = '';
  if (videoResult?.data?.resultJson) {
    try {
      const parsed = JSON.parse(videoResult.data.resultJson);
      videoUrl = (parsed.resultUrls || parsed.originUrls || [])[0] || '';
    } catch(e) { videoUrl = ''; }
  }
  if (!videoUrl) videoUrl = videoResult?.video?.url || videoResult?.data?.url || '';
  const dialogueText = $('세그먼트 분리').all()[i]?.json?.dialogue || '';

  if (!videoUrl) throw new Error('세그먼트 ' + (i+1) + ' 영상 URL 없음');

  segments.push({
    index: i + 1, videoUrl, duration: CLIP_DURATION,
    start: currentStart, dialogueText,
  });
  currentStart += CLIP_DURATION;
}

const totalDuration = currentStart;

// === SRT 자막 생성 ===
function toSrtTime(sec) {
  const h = Math.floor(sec / 3600);
  const m = Math.floor((sec % 3600) / 60);
  const s = Math.floor(sec % 60);
  const ms = Math.round((sec - Math.floor(sec)) * 1000);
  return String(h).padStart(2, '0') + ':' +
         String(m).padStart(2, '0') + ':' +
         String(s).padStart(2, '0') + ',' +
         String(ms).padStart(3, '0');
}

let srtLines = [];
for (let i = 0; i < segments.length; i++) {
  const seg = segments[i];
  const startTime = toSrtTime(seg.start);
  const endTime = toSrtTime(seg.start + seg.duration);
  srtLines.push((i + 1) + '\n' + startTime + ' --> ' + endTime + '\n' + seg.dialogueText + '\n');
}
const srtContent = srtLines.join('\n');

// === NCA inputs ===
let inputs = [];
let inputIndex = 0;
const videoInputs = [];

for (const seg of segments) {
  inputs.push({ file_url: seg.videoUrl, options: [] });
  videoInputs.push(inputIndex);
  inputIndex++;
}

// === filter_complex (영상+오디오 concat만, drawtext 없음) ===
let filterParts = [];

for (let i = 0; i < segments.length; i++) {
  const vIdx = videoInputs[i];
  filterParts.push(
    '[' + vIdx + ':v]trim=duration=' + segments[i].duration +
    ',setpts=PTS-STARTPTS' +
    ',scale=1920:1080:force_original_aspect_ratio=decrease' +
    ',pad=1920:1080:(ow-iw)/2:(oh-ih)/2:black' +
    ',fps=30[v' + i + ']'
  );
  filterParts.push(
    '[' + vIdx + ':a]atrim=duration=' + segments[i].duration +
    ',asetpts=PTS-STARTPTS[a' + i + ']'
  );
}

// 비디오 concat
const vLabels = segments.map((_, i) => '[v' + i + ']').join('');
filterParts.push(vLabels + 'concat=n=' + segments.length + ':v=1:a=0[vfinal]');

// 오디오 concat
const aLabels = segments.map((_, i) => '[a' + i + ']').join('');
filterParts.push(aLabels + 'concat=n=' + segments.length + ':v=0:a=1[afinal]');

const filter_complex = filterParts.join(';');

const nca_payload = {
  inputs, filters: [{ filter: filter_complex }],
  outputs: [{
    options: [
      { option: "-map", argument: "[vfinal]" },
      { option: "-map", argument: "[afinal]" },
      { option: "-c:v", argument: "libx264" },
      { option: "-preset", argument: "medium" },
      { option: "-crf", argument: "18" },
      { option: "-c:a", argument: "aac" },
      { option: "-b:a", argument: "256k" },
      { option: "-pix_fmt", argument: "yuv420p" },
      { option: "-movflags", argument: "+faststart" },
      { option: "-t", argument: String(totalDuration) }
    ],
    file_name: "story_shorts_final.mp4"
  }]
};

return [{
  json: {
    nca_payload, totalDuration, srtContent,
    segmentCount: segments.length, hookTitle,
    subject: $('콘텐츠 파싱').first().json.Subject,
    caption: $('콘텐츠 파싱').first().json.Caption,
    comment: $('콘텐츠 파싱').first().json.Comment,
  }
}];
```
