- 타입: n8n-nodes-base.code

```javascript
const raw = $input.first().json;
let text = '';
if (raw.content && raw.content.parts) {
  text = raw.content.parts[0].text;
} else if (typeof raw === 'string') {
  text = raw;
} else if (raw.text) {
  text = raw.text;
} else {
  text = JSON.stringify(raw);
}

const cleanText = text.replace(/```json\n?/g, '').replace(/```\n?/g, '').trim();
const data = JSON.parse(cleanText);

if (!data.scenes || data.scenes.length < 3) {
  throw new Error('scenes must have at least 3 items, got: ' + (data.scenes?.length || 0));
}

// dialogue_turns 검증
for (let i = 0; i < data.scenes.length; i++) {
  const scene = data.scenes[i];
  if (!scene.dialogue_turns || scene.dialogue_turns.length < 2) {
    if (scene.dialogue && scene.speaker) {
      scene.dialogue_turns = [{speaker: scene.speaker, line: scene.dialogue}];
    } else {
      throw new Error('scene ' + (i+1) + ' missing dialogue_turns');
    }
  }
  if (!scene.dialogue) {
    scene.dialogue = scene.dialogue_turns.map(t => t.line).join(' ');
  }
  if (!scene.speaker) {
    scene.speaker = scene.dialogue_turns[0].speaker;
  }
}

return [{
  json: {
    hook_title: data.hook_title,
    scenes: data.scenes,
    Subject: data.Subject,
    Caption: data.Caption,
    Comment: data.Comment,
    BGM_prompt: data.BGM_prompt,
    fullDialogue: data.scenes.map(s => s.dialogue).join(' '),
    segmentCount: data.scenes.length,
    Status: '준비',
    generatedAt: new Date().toISOString()
  }
}];
```

```javascript
const data = $('콘텐츠 파싱').first().json;
const scenes = data.scenes || [];

// 고정 참조 이미지 URL
const refResults = $('참조 이미지 프롬프트').all();
const refA_url = refResults[0]?.json?.refUrl || '';
const refB_url = refResults[1]?.json?.refUrl || '';

if (!refA_url || !refB_url) {
  throw new Error('참조 이미지 URL 없음! refA=' + !!refA_url + ', refB=' + !!refB_url);
}

// Veo3 핑퐁 대화 프롬프트 생성
function buildVeo3Dialogue(turns) {
  const speakerDesc = {
    'A': 'The first man (short fade haircut, black hoodie, gold chain)',
    'B': 'The second man (medium twists hairstyle, white graphic t-shirt)'
  };

  return turns.map((turn, i) => {
    const desc = speakerDesc[turn.speaker] || speakerDesc['A'];
    const line = turn.line.replace(/"/g, "'");
    if (i === 0) return desc + ' says "' + line + '"';
    return 'then ' + desc + ' responds "' + line + '"';
  }).join(', ');
}

return scenes.map((scene, i) => {
  const speaker = scene.speaker || scene.dialogue_turns?.[0]?.speaker || 'A';

  const cleanImagePrompt = scene.image_prompt.replace(/"/g, "'");

  // dialogue_turns 핑퐁 프롬프트
  let dialoguePart;
  if (scene.dialogue_turns && scene.dialogue_turns.length > 1) {
    dialoguePart = buildVeo3Dialogue(scene.dialogue_turns);
  } else {
    const cleanDialogue = scene.dialogue.replace(/"/g, "'");
    dialoguePart = 'The person is naturally speaking in Korean, saying: ' + cleanDialogue;
  }

  const veo3_prompt = cleanImagePrompt +
    '. ' + dialoguePart +
    '. Natural back-and-forth conversation with realistic timing. Natural lip synchronization for both speakers, subtle body gestures while talking, cinematic camera movement.';

  return {
    json: {
      index: i,
      dialogue: scene.dialogue,
      dialogue_turns: scene.dialogue_turns || [{speaker: speaker, line: scene.dialogue}],
      image_prompt: scene.image_prompt,
      veo3_prompt: veo3_prompt,
      speaker: speaker,
      refUrl_A: refA_url,
      refUrl_B: refB_url,
    }
  };
});
```

### NCA 데이터 준비
- 타입: n8n-nodes-base.code
- 코드:
```javascript
// NCA 스토리형 숏츠 - Veo3 Fast (audio on) - 가로(16:9) 버전
const hookTitle = $('콘텐츠 파싱').first().json.hook_title;
const segmentCount = $('콘텐츠 파싱').first().json.segmentCount;

// === 세그먼트 데이터 수집 ===
let segments = [];
let currentStart = 0;
const CLIP_DURATION = 8; // Veo3 8초 클립

for (let i = 0; i < segmentCount; i++) {
  const videoResult = $('영상 완료 체크').all()[i]?.json;
  let videoUrl = '';
  // Kie.ai Veo3 응답: data.response.resultUrls
  if (videoResult?.data?.response?.resultUrls) {
    videoUrl = videoResult.data.response.resultUrls[0] || '';
  }
  if (!videoUrl && videoResult?.data?.resultJson) {
    try {
      const parsed = JSON.parse(videoResult.data.resultJson);
      videoUrl = (parsed.resultUrls || parsed.originUrls || [])[0] || '';
    } catch(e) { videoUrl = ''; }
  }
  if (!videoUrl) videoUrl = videoResult?.video?.url || videoResult?.data?.url || '';
  const segData = $('세그먼트 분리').all()[i]?.json;
  const dialogueText = segData?.dialogue || (segData?.dialogue_turns || []).map(t => t.line).join(' ') || '';

  if (!videoUrl) throw new Error('세그먼트 ' + (i+1) + ' 영상 URL 없음');

  segments.push({
    index: i + 1, videoUrl, duration: CLIP_DURATION,
    start: currentStart, dialogueText,
  });
  currentStart += CLIP_DURATION;
}

const totalDuration = currentStart;

// === SRT 자막 생성 ===
function toSrtTime(sec) {
  const h = Math.floor(sec / 3600);
  const m = Math.floor((sec % 3600) / 60);
  const s = Math.floor(sec % 60);
  const ms = Math.round((sec - Math.floor(sec)) * 1000);
  return String(h).padStart(2, '0') + ':' +
         String(m).padStart(2, '0') + ':' +
         String(s).padStart(2, '0') + ',' +
         String(ms).padStart(3, '0');
}

let srtLines = [];
for (let i = 0; i < segments.length; i++) {
  const seg = segments[i];
  const startTime = toSrtTime(seg.start);
  const endTime = toSrtTime(seg.start + seg.duration);
  srtLines.push((i + 1) + '\n' + startTime + ' --> ' + endTime + '\n' + seg.dialogueText + '\n');
}
const srtContent = srtLines.join('\n');

// === NCA inputs ===
let inputs = [];
let inputIndex = 0;
const videoInputs = [];

for (const seg of segments) {
  inputs.push({ file_url: seg.videoUrl, options: [] });
  videoInputs.push(inputIndex);
  inputIndex++;
}

// === filter_complex (영상+오디오 concat만, drawtext 없음) ===
let filterParts = [];

for (let i = 0; i < segments.length; i++) {
  const vIdx = videoInputs[i];
  filterParts.push(
    '[' + vIdx + ':v]trim=duration=' + segments[i].duration +
    ',setpts=PTS-STARTPTS' +
    ',scale=1920:1080:force_original_aspect_ratio=decrease' +
    ',pad=1920:1080:(ow-iw)/2:(oh-ih)/2:black' +
    ',fps=30[v' + i + ']'
  );
  filterParts.push(
    '[' + vIdx + ':a]atrim=duration=' + segments[i].duration +
    ',asetpts=PTS-STARTPTS[a' + i + ']'
  );
}

// 비디오 concat
const vLabels = segments.map((_, i) => '[v' + i + ']').join('');
filterParts.push(vLabels + 'concat=n=' + segments.length + ':v=1:a=0[vfinal]');

// 오디오 concat
const aLabels = segments.map((_, i) => '[a' + i + ']').join('');
filterParts.push(aLabels + 'concat=n=' + segments.length + ':v=0:a=1[afinal]');

const filter_complex = filterParts.join(';');

const nca_payload = {
  inputs, filters: [{ filter: filter_complex }],
  outputs: [{
    options: [
      { option: "-map", argument: "[vfinal]" },
      { option: "-map", argument: "[afinal]" },
      { option: "-c:v", argument: "libx264" },
      { option: "-preset", argument: "medium" },
      { option: "-crf", argument: "18" },
      { option: "-c:a", argument: "aac" },
      { option: "-b:a", argument: "256k" },
      { option: "-pix_fmt", argument: "yuv420p" },
      { option: "-movflags", argument: "+faststart" },
      { option: "-t", argument: String(totalDuration) }
    ],
    file_name: "story_shorts_final.mp4"
  }]
};

return [{
  json: {
    nca_payload, totalDuration, srtContent,
    segmentCount: segments.length, hookTitle,
    subject: $('콘텐츠 파싱').first().json.Subject,
    caption: $('콘텐츠 파싱').first().json.Caption,
    comment: $('콘텐츠 파싱').first().json.Comment,
  }
}];
```
